Define AWS IAM policies with ease using a smaller, quicker-to-read format.

policy.iam 
ALLOW dynamodb:Batch*, dynamodb:DeleteItem
  ON dynamodb:::table/push*;

ALLOW s3:PutRecord ON kinesis:::stream/push*;
policy.json 
[
  {
    "Effect": "Allow",
    "Action": [
      "dynamodb:Batch*",
      "dynamodb:DeleteItem"
    ],
    "Resource": [
      "arn:aws:dynamodb:::table/push*"
    ]
  },
  {
    "Effect": "Allow",
    "Action": [
      "s3:PutRecord"
    ],
    "Resource": [
      "arn:aws:kinesis:::stream/push*"
    ]
  }
]
